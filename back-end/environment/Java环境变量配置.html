# Java环境变量配置

# windows环境

# jdk8

  1. JAVA_HOME
E:\Program Files x86\Java\Java-8u351
  1. CLASSPATH
.;%JAVA_HOME%\lib\tools.jar;%JAVA_HOME%\lib\dt.jar
  1. Path添加
;%JAVA_HOME%\bin;%JAVA_HOME%\jre\bin

注意看分号

# jdk9+

JAVA_HOME

E:\Program Files x86\Java\Java12\jdk12.0.2

# 常用命令

  1. 启动程序
java -jar app-name.jar
  1. 关闭程序
# 查看端口号
netstat -ano | findstr "8080"
# 关闭端口
taskkill -pid <进程ID> -f
  1. 后台运行
javaw -jar app-name.jar

# Linux环境

# debain系统

# 准备JDK安装包

jdk-8u161-linux-x64.tar.gz ，放在root目录下

# 卸载系统自带JDK

rpm -qa | grep java
yum -y remove java-...

# 创建目录并解压

cd /usr/local/
mkdir java
cd java
tar -zxvf /root/jdk-8u161-linux-x64.tar.gz -C ./

# 配置JDK环境变量

编辑/etc/profile文件，文件尾部添加JDK配置

JAVA_HOME=/usr/local/java/jdk1.8.0_161
CLASSPATH=$JAVA_HOME/lib/
PATH=$PATH:$JAVA_HOME/bin
export PATH JAVA_HOME CLASSPATH

让环境变量生效

source /etc/profile

# 验证

java -version
javac
