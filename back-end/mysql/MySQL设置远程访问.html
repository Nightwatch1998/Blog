# MySQL Server添加远程主机访问权限

首先登录mysql，查看所有用户和主机信息

use mysql;
select Host, User from user;

增加一个新的用户及地址

create user 'username'@'%' identified by 'password';
grant all on *.* to 'username'@'%';
